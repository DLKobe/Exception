                 $(document).ready(function () {
    document.getElementById("btn").onclick=
        function () {

            $.ajax({
                  type:"GET",
                  url:"/showchart/",

                  dataType:"json",
                  success:function (data,status) {
                      console.log(data)
                      var d=data["data"]
                      a=d[0][0]
                      b=d[1][0]
                      c=d[2][0]
                      k=d[3][0]
                      e=d[4][0]

                      f=d[0][1]
                      g=d[1][1]
                      h=d[2][1]
                      o=d[3][1]
                      j=d[4][1]
                      var myChart = echarts.init(document.getElementById('main'));
                      option = {
                          title: {
                              text: 'TOP5'
                          },
                          xAxis: {
                              name:'名称',
                              type: 'category',
                              data: [a,b,c,k,e]
                          },
                          yAxis: {
                              name:'次数',
                              type: 'value'
                          },
                          series: [{
                              data: [f, g, h, o,j],
                              type: 'bar'
                          }]
                      };
                      myChart.setOption(option);







                  }

              })



        }

})






    </script>
